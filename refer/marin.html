CSS margin

margin-top: 10;
margin-right: 11px;
margin-bottom: 12px;
margin-left: 13px;
margin: 10px;         /* 위, 오른쪽, 아래, 왼쪽: 10px */
margin: 10px 11px;       /* 위, 아래: 10px 왼쪽, 오른쪽: 11px */
margin: 10px 11px 12px;      /* 위:10px, 왼쪽, 오른쪽: 11px 아래: 12px */
margin: 10px 11px 12px 13px;       /* 위:10px, 오른쪽: 11px, 아래: 12px, 왼쪽: 13px (padding)도 같다.*/
margin: auto;       /* 자동값, 기본값 (padding)은 없다.*/
margin: 0 auto;        /* 가운데 정렬(블록구조) */
padding: 10px;       /* 위, 오른쪽, 아래, 왼쪽: 10px */
padding: 10px 11px;      /* 위, 아래: 10px 왼쪽, 오른쪽: 11px */
padding: 10px 11px 12px;       /* 위:10px, 왼쪽, 오른쪽: 11px 아래: 12px*/
padding: 10px 11px 12px 13px;       /* 위:10px, 오른쪽: 11px, 아래: 12px, 왼쪽: 13px (padding)도 같다.*/
