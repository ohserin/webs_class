1.1 2단부터 9단까지 출력하시오!

for(let i=2; i<=9; i++){
    document.write("
");

    for(let j=1; j<=9; j++){
        let sum = i * j;

        document.write( i + " * " + j + " = " + sum, "
" );
    }
}

1.2 while문을 이용하여 2단부터 9단까지 출력하시오!

var i = 2;
        do {
            document.write(" " + i + "단:");
            var j = 1;
            do {
                document.write(+i + "*" + j + "=" + i * j);
                document.write(" , ");
                j++;
            } while (j <= 9);
            document.write();
            i++;
        } while (i <= 9);
