1~25까지 숫자를 넣으시오!

{
     let num = 1;
     let table = "<table>";
         
     for ( let i=1; i<=5; i++ ){
         table += "<tr>";
             for ( let j=1; j<=5; j++ ){
                 table += "<td>"+ num +"</td>";
                 num++;
             }
         table += "</tr>";   
     }
     table += "</table>";
 
     document.querySelector(".result02").innerHTML = table;
 }
